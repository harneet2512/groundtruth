// src/daemon/validation-engine.ts
class ValidationEngine {
  async validateCode(code: string, file: string) {
    const errors = [];
    
    // Extract imports
    const imports = this.parseImports(code);
    
    // Validate each
    for (const imp of imports) {
      const valid = await this.lsp.validateImport(file, imp.name);
      if (!valid) {
        errors.push({
          type: 'invalid_import',
          name: imp.name,
          suggestion: await this.findClosestMatch(imp.name)
        });
      }
    }
    
    return { valid: errors.length === 0, errors };
  }
}

interface CachedSuggestion {
  errorHash: string;        // Hash of the error
  suggestion: string;       // AI's suggestion
  confidence: number;       // 0-1
  usageCount: number;       // How many times used
  lastUsed: Date;
  tokensCost: number;
}

class AICache {
  async get(error: ValidationError): Promise<CachedSuggestion | null> {
    // Create hash of error (ignores variable names, line numbers)
    const hash = this.hashError(error);
    
    // Check cache
    const cached = await this.db.query(
      'SELECT * FROM ai_cache WHERE error_hash = ? AND confidence > 0.8',
      [hash]
    );
    
    if (cached) {
      // Update usage stats
      await this.db.query(
        'UPDATE ai_cache SET usage_count = usage_count + 1, last_used = ? WHERE error_hash = ?',
        [new Date(), hash]
      );
      
      return cached;
    }
    
    return null;
  }
  
  private hashError(error: ValidationError): string {
    // Normalize error (remove specifics, keep pattern)
    const normalized = {
      type: error.type,
      pattern: this.extractPattern(error.code),
      availableOptions: error.availableOptions.sort()
    };
    
    return crypto.createHash('sha256')
      .update(JSON.stringify(normalized))
      .digest('hex');
  }
}

class TokenBudgetManager {
  private config = {
    dailyLimit: 50000,      // 50k tokens/day (~$0.10/day)
    perValidation: 1000,    // Max tokens per AI call
    warningThreshold: 0.8   // Warn at 80%
  };
  
  async canUseTokens(estimatedTokens: number): Promise<boolean> {
    const today = await this.getTodayUsage();
    
    // Check if we'd exceed limit
    if (today.tokens + estimatedTokens > this.config.dailyLimit) {
      return false;
    }
    
    // Warn user if approaching limit
    const percentage = today.tokens / this.config.dailyLimit;
    if (percentage > this.config.warningThreshold) {
      logger.warn(`AI budget at ${(percentage * 100).toFixed(0)}%`);
    }
    
    return true;
  }
  
  async trackUsage(tokens: number, cost: number) {
    await this.db.query(
      'INSERT INTO token_usage (date, tokens, cost) VALUES (?, ?, ?)',
      [new Date().toISOString().split('T')[0], tokens, cost]
    );
  }
  
  async getTodayStats() {
    const stats = await this.db.query(
      'SELECT SUM(tokens) as tokens, SUM(cost) as cost FROM token_usage WHERE date = ?',
      [new Date().toISOString().split('T')[0]]
    );
    
    return {
      tokens: stats.tokens || 0,
      cost: stats.cost || 0,
      remaining: this.config.dailyLimit - (stats.tokens || 0),
      percentage: ((stats.tokens || 0) / this.config.dailyLimit * 100).toFixed(1)
    };
  }
}

  /**
   * Use LSP to validate EVERY import/function call
   * NO AI needed, 100% accurate, instant
   */
  
  async validateCode(code: string, filePath: string): Promise<ValidationResult> {
    const errors: Error[] = [];
    
    // Parse the code
    const ast = parseTypeScript(code);
    
    // 1. Validate ALL imports
    for (const imp of ast.imports) {
      const available = await this.lsp.getExports(imp.source);
      
      if (!available.includes(imp.name)) {
        errors.push({
          type: 'hallucinated_import',
          line: imp.line,
          message: `${imp.name} doesn't exist in ${imp.source}`,
          available: available,  // Show what ACTUALLY exists
          confidence: 1.0  // 100% sure, we checked LSP
        });
      }
    }
    
    // 2. Validate function calls
    for (const call of ast.functionCalls) {
      const definition = await this.lsp.getDefinition(call.name);
      
      if (!definition) {
        errors.push({
          type: 'undefined_function',
          line: call.line,
          message: `${call.name} is not defined`,
          confidence: 1.0
        });
        continue;
      }
      
      // 3. Validate function signature
      const expectedArgs = definition.parameters.length;
      const providedArgs = call.arguments.length;
      
      if (expectedArgs !== providedArgs) {
        errors.push({
          type: 'wrong_signature',
          line: call.line,
          message: `${call.name} expects ${expectedArgs} args, got ${providedArgs}`,
          expected: definition.signature,
          confidence: 1.0
        });
      }
    }
    
    // 4. Validate types
    for (const assignment of ast.assignments) {
      const inferredType = await this.lsp.getType(assignment.value);
      const expectedType = await this.lsp.getType(assignment.variable);
      
      if (!this.typesMatch(inferredType, expectedType)) {
        errors.push({
          type: 'type_mismatch',
          line: assignment.line,
          message: `Expected ${expectedType}, got ${inferredType}`,
          confidence: 1.0
        });
      }
    }
    
    return {
      valid: errors.length === 0,
      errors: errors
    };
  }

// src/lsp-validator.ts

import * as lsp from 'vscode-languageserver-node';
import * as fs from 'fs';
import * as path from 'path';

export class LSPValidator {
  private connection: lsp.Connection;
  
  async init(projectPath: string) {
    // Start TypeScript LSP
    const serverPath = require.resolve('typescript-language-server/lib/cli.js');
    
    const serverModule = require('child_process').spawn(
      'node',
      [serverPath, '--stdio'],
      { cwd: projectPath }
    );
    
    this.connection = lsp.createConnection(
      new lsp.StreamMessageReader(serverModule.stdout),
      new lsp.StreamMessageWriter(serverModule.stdin)
    );
    
    await this.connection.listen();
  }
  
  async validateImport(
    importName: string,
    importPath: string,
    currentFile: string
  ): Promise<{ valid: boolean; available?: string[] }> {
    // Resolve absolute path
    const absolutePath = path.resolve(
      path.dirname(currentFile),
      importPath
    );
    
    // Get exports from LSP
    const uri = `file://${absolutePath}`;
    const documentSymbols = await this.connection.sendRequest(
      'textDocument/documentSymbol',
      { textDocument: { uri } }
    );
    
    // Extract exported symbols
    const exports = documentSymbols
      .filter((s: any) => s.kind === lsp.SymbolKind.Function || s.kind === lsp.SymbolKind.Variable)
      .map((s: any) => s.name);
    
    return {
      valid: exports.includes(importName),
      available: exports
    };
  }
}

// Add to mcp-server.ts

server.setRequestHandler('tools/call', async (request) => {
  if (request.params.name === 'validate_and_fix') {
    const { code, file } = request.params.arguments;
    
    // Parse imports from code
    const imports = parseImports(code);
    
    const fixes = [];
    
    for (const imp of imports) {
      const result = await validator.validateImport(
        imp.name,
        imp.path,
        file
      );
      
      if (!result.valid) {
        // Find closest match
        const closest = findClosestMatch(imp.name, result.available);
        
        fixes.push({
          line: imp.line,
          original: imp.name,
          suggested: closest,
          reason: `${imp.name} doesn't exist, did you mean ${closest}?`
        });
      }
    }
    
    return {
      content: [{
        type: 'text',
        text: JSON.stringify(fixes, null, 2)
      }]
    };
  }
});

function findClosestMatch(target: string, options: string[]): string {
  // Simple Levenshtein distance
  let closest = options[0];
  let minDistance = Infinity;
  
  for (const option of options) {
    const distance = levenshtein(target, option);
    if (distance < minDistance) {
      minDistance = distance;
      closest = option;
    }
  }
  
  return closest;
}
